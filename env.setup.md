# FOR LINUX/MAC 
### Installing venv 
    sudo apt-get install 
    python3.6-venv 
### Creating virtual env 
    python3 -m venv env 
### Activating virtual env 
    source env/bin/activate

# FOR WINDOWS 
### installing venv 
    pip install virtualenv 
### Creating virtual env 
    python -m venv env 
### Activating virtual env 
    .\env\Scripts\activate
